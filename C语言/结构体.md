结构体是由基本数据类型构成的、并用一个标识符来命名的各种变量的组合。

定义结构变量的一般格式为:

```c
struct 结构名
{
类型变量名;
类型变量名;
...
} 结构变量;
```
```
1. #include<stdio.h>
2. //定义一个名为的Student 的结构体，并且创建一个变量名student
3. struct Student{
4. int age;
5. float score;
6. char* name;
7. }student;
8. main(){
9. //创建一个变量为s 的结构体对象，并赋值
10. struct Student s ={23,89.5,"张三"};
11. //给结构体变量student 赋值
12. student.age=24;
13. student.score=80;
14. student.name="李四";
15. printf("张三的个人信息：\n");
16. //调用结构体对象的属性
17. printf("年龄=%d\n 分数=%f\n 姓名=%s\n",s.age,s.score,s.name);
18. printf("李四的个人信息:\n");
19. printf("年龄=%d\n 分数=%f\n 姓名=%s\n",student.age,student.score,student.name);
```
